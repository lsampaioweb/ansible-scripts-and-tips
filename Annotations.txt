- name: Deploy a web application
  hosts: db_and_web_server
  tasks:
    - name: Install MySQL database
      apt:
        name: "{{ item }}"
        state:  "present"
      loop:
        - mysql-server
        - mysql-client

    - name: Start mysql
      service:
        name: "mysql"
        enabled: true
        state: "started"
    
    - name: Install Python Flask dependencies
      pip:
        name: '{{ item }}'
        state: present
      loop:
       - "flask"
       - "flask-mysql"

    - name: Copy web-server code
      copy: 
        src: "app.py"
        dest: "/opt/app.py"
